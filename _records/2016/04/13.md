# go的defer、closure、method binding
```golang
package main

import "fmt"

type number int

func (n number) print()   { fmt.Println(n) }
func (n *number) pprint() { fmt.Println(*n) }

func main() {
	var n number
	defer n.print()
	defer n.pprint()
	defer func() { n.print() }()
	defer func() { n.pprint() }()
	defer func(x number) { x.print() }(n)
	defer func(x number) { x.pprint() }(n)
	defer func(x *number) { x.pprint() }(&n)

	n = 3
}
```
输出如下
```
3
0
0
3
3
3
0
```
